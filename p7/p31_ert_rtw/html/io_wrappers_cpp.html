  140       extern "C" uint8_t MW_EthernetAndUDPBegin(uint8_t udpindex, uint32_t localPort)
  141       {       
  142           static int only_one_ethernet_begin;
  143           if (only_one_ethernet_begin==0)
  144           {
  145             #if (defined(_RTT_DISABLE_Ethernet_DHCP_) && (_RTT_DISABLE_Ethernet_DHCP_ !=0))
  146               Ethernet.begin(mac, localIpAddress);
  147               configureSuccess = 1;
  148             #else
  149               configureSuccess = Ethernet.begin(mac);
  150             #endif
  151               only_one_ethernet_begin++;
  152               
  153               for(trialcount=0;trialcount<5;trialcount++)
  154               {
  155                   if(Serial)
  156                   {
  157                       break;
  158                   }else
  159                   {
  160                       delay(1000);
  161                   }
  162               }
  163               // Depending on if the IP address assignment using DHCP has been successful or not,
  164               // send the appropriate information back to the host.
  165               // If External mode over TCP/IP or Wifi, do not print this Info. Information of IP address will be printed as 
  166               // a part of rtiostream_open
  167             #if (!(defined(_RTT_TCPIP_EXTMODE_) && (_RTT_TCPIP_EXTMODE_ == 1)))
  168                   if (configureSuccess==1)
  169                   {
  170                      if(Ethernet.localIP()[0] == 0 || Ethernet.localIP()[0] == 255)
  171                       {
  172                           /*Tried to assign Static IP, failed due to one of the following reasons :
  173                            * 1. Ethernet shield is not connected properly with Arduino.
  174                            * 2. Ethernet shield is not connected to a network.
  175                            **/
  176                           Serial.println("<<<IP address: Failed to configure.>>>");
  177                       }
  178                       else
  179                       {
  180                           // If the Configuration is successful, relay back the assigned IP address.
  181                           Serial.print("<<<IP address: ");
  182                           for (byte thisByte = 0; thisByte < 4; thisByte++)
  183                           {
  184                               // Print the value of each byte of the IP address:
  185                               Serial.print(Ethernet.localIP()[thisByte], DEC);
  186                               if (thisByte<3)
  187                               {
  188                                   Serial.print(".");
  189                               }
  190                           }
  191                           Serial.println(">>>");
  192                       }     
  193                   }
  194                  else
  195                  {
  196                      // If the Configuration failed,relay back the error message.
  197                      Serial.println("<<<IP address: Failed to configure.>>>");
  198                  }
  199                  Serial.flush();
  200             #endif
  201           }
  202           return (Udp[udpindex].begin(localport[udpindex]));
  203       }

  205       extern "C" int MW_UDPFinalWrite(uint8_t udpindex, uint8_t data, uint8_t remoteIPOctect1,uint8_t remoteIPOctect2,uint8_t remoteIPOctect3,uint8_t remoteIPOctect4, uint32_t remoteport)
  206       {
  207           IPAddress remoteIpAddress(remoteIPOctect1,remoteIPOctect2, remoteIPOctect3, remoteIPOctect4);
  208           Udp[udpindex].beginPacket(remoteIpAddress, remoteport);
  209           Udp[udpindex].write(data);
  210           return(Udp[udpindex].endPacket());
  211       }

  362   extern "C" void MW_delay_in_milliseconds(uint32_T period)
  363   {
  364       uint32_T start_time = 0;
  365       start_time = millis();
  366       while(millis()- start_time <= period) 
  367       {
  368           1;
  369       }
  370   }
